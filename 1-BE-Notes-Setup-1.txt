1. create <folder>
2. cd folder
3. npm init
4. npm i express dotenv nodemon
5. create .env and .gitignore files

6. in .gitignore file add the ff:
node_modules
.env

7. .env file add the ff:
PORT=8000

8. create app.js file.
9 add these line of codes on app.js
// import express application
const express = require('express')

//initialize express app
const app = express()

//import dotenv
require('dotenv').config()

app.get('/', (req, res) => {
    res.send('hello from node!');
})

// import port from .env file
const port = process.env.PORT || 8000

// run application
app.listen(port, () => {
    console.log(`Server is running on port ${port}`)
});

10. run application 
- $ npm app.js

11. on local server to prevent from stopping and rerunning start command,
we add nodemon on start script on package.json
"scripts": {
    "start": "nodemon app.js",
    "test": "echo \"Error: no test specified\" && exit 1"
},

12. run application 
- $ npm start
