1. install mongoose for mongodb
- npm i mongoose

2. import mongoose on app.js
const mongoose = require('mongoose')

mongoose
    .connect(process.env.DATABASE)
    .then(result => {
        console.log('DB Connected)
    })
    .catch(err => console.log(err))

3. on .env file add DATABASE
DATABASE=mongodb://localhost:27017/ecommerce-app
