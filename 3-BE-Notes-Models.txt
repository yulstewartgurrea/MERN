2. initialize mongoose schema
3. create schema
4. export schema

// import and initialize mongoose
const mongoose = require('mongoose');

// initialize mongoose schema
const Schema = mongoose.Schema;

// create schema
const userSchema = new Schema({
    firstname: {
        type: String,
        required: true
    },
    lastname: {
        type: String,
        required: true
    },
    middlename: {
        type: String,
        required: false
    },
    birthday: {
        type: String,
        required: false
    },
    address: {
        type: String,
        required: false
    }
}, { timestamps: true});

// export schema
module.exports = mongoose.model('user', userSchema)
